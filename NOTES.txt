To get started run the following SQL commands:
USE calendar1; (Or select calendar1 in phpMyAdmin)

CREATE DATABASE calendar1;


GRANT ALL ON calendar1.* TO 'fred'@'localhost' IDENTIFIED BY 'zap';
GRANT ALL ON calendar1.* TO 'fred'@'127.0.0.1' IDENTIFIED BY 'zap';



CREATE TABLE users (
   user_id INTEGER NOT NULL
     AUTO_INCREMENT KEY,
   username VARCHAR(128),
   email VARCHAR(128),
   password VARCHAR(128),
   INDEX(email)
) ENGINE=InnoDB CHARSET=utf8;


CREATE TABLE events (
  event_id INTEGER NOT NULL PRIMARY KEY AUTO_INCREMENT,
  eventname VARCHAR(255),
  eventdate datetime,
  eventnote VARCHAR(300)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
